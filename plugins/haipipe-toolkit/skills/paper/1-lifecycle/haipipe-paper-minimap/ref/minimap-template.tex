% =====================================================================
% minimap-template.tex  --  VENUE-NEUTRAL minimap scaffold
% =====================================================================
% This file defines the LaTeX CONTENT + MACROS a 5-minimap.tex should cover.
% It is venue-NEUTRAL: the SHAPE (which sections, abstract form, hypotheses or
% not, page budget) comes from _venue/playbook-<venue> (its "-> Minimap"
% mapping). Copy this to <paper>/0-lifecycle/5-minimap/5-minimap.tex, then fill
% the <...> placeholders and reorder the sections to the pinned venue.
%
% WHAT A MINIMAP COVERS  (content checklist; see CHECKLIST.md for the done-gate):
%   [ ] short \title = the paper title only, SHAPED TO THE VENUE (no subtitle)
%   [ ] one intro note (what the minimap is) + a Style Contract box
%   [ ] Abstract, in the venue's abstract shape (unstructured prose / structured)
%   [ ] one block per manuscript section, IN THE VENUE'S SECTION ORDER, each:
%         - \pspine paragraph head  (label | one-line job)
%         - 4-5 sentence-points (spts): one point = one manuscript sentence,
%           tagged with the claim it carries via \cl{Cn}
%         - \dcall + \input the real display float where first discussed (thumbnail)
%         - one \nnote: LEAN (one clause), NO "verb, role." editorial prefix
%         - \pfb advisor-feedback line(s) below the note, with a resolution status
%   [ ] Supplement (eAppendix) SYNCED to 0-sections/{A..}*.tex + 0-Supplementary-*
%   [ ] Write-Time Reconciliations (the \wt items, consolidated)
%   NO Coverage Check crosswalk by default -> run it as a SILENT lint instead.
% =====================================================================
\documentclass[11pt]{article}
\usepackage[margin=0.85in]{geometry}
\usepackage{xcolor}
\usepackage{amsmath}      % \text in display-table math cells
\usepackage{amssymb}      % \checkmark etc. in display tables
\usepackage{enumitem}
\usepackage{booktabs}\usepackage{tabularx}\usepackage{array}
\usepackage{graphicx}\usepackage{caption}\usepackage{float}
\usepackage{hyperref}

% ---- inline display thumbnails: each \input'd float -> boxed + shrunk -------
\newcommand{\mmscale}{0.5}
\makeatletter
\newsavebox{\mmbox}
\newenvironment{mmframe}[1]{%
  \def\@captype{#1}\par\smallskip
  \begin{lrbox}{\mmbox}\begin{minipage}{\linewidth}\centering\scriptsize}%
 {\end{minipage}\end{lrbox}%
  \begingroup\centering\setlength{\fboxsep}{4pt}\setlength{\fboxrule}{0.6pt}%
  \fbox{\scalebox{\mmscale}{\usebox{\mmbox}}}\par\endgroup\smallskip}
\renewenvironment{figure}[1][]{\mmframe{figure}}{\endmmframe}
\renewenvironment{table}[1][]{\mmframe{table}}{\endmmframe}
\makeatother
\setlength{\abovecaptionskip}{3pt}\captionsetup{font=small}

% ---- colors + tags ---------------------------------------------------------
\definecolor{cTodo}{rgb}{0.80,0.0,0.0}
\definecolor{cClaim}{rgb}{0.0,0.35,0.65}
\definecolor{cNote}{gray}{0.40}
\definecolor{cFb}{rgb}{0.55,0.0,0.15}
\newcommand{\wt}[1]{\textcolor{cTodo}{\textbf{[WRITE-TIME]}~#1}}

% ---- claim tag: shows the index AND a short CONTENT gloss (never bare) ------
% Seed clg@Cn from the claim titles in 2-claims.tex so the tag is self-explanatory
% and the minimap stays consistent with the ledger. Multi-claim tags: spell the
% glosses out, e.g. \cl{C1: <a>, C2: <b>}.
\makeatletter
\newcommand{\cl@gloss}[1]{\ifcsname clg@#1\endcsname\csname clg@#1\endcsname\else #1\fi}
\@namedef{clg@C0}{C0: <data anchor>}
\@namedef{clg@C1}{C1: <primary, one phrase>}
\@namedef{clg@C2}{C2: <one phrase>}
\@namedef{clg@C3}{C3: <one phrase>}
\newcommand{\cl}[1]{\,\textcolor{cClaim}{\footnotesize\textbf{[\cl@gloss{#1}]}}}
\makeatother

\newcommand{\dcall}[1]{\par\smallskip\noindent\colorbox{gray!18}{\parbox{0.96\linewidth}{%
  \footnotesize $\blacktriangleright$~\textbf{Display here:}~#1}}\par\smallskip}
\newcommand{\scite}[1]{\par\smallskip\noindent{\footnotesize $\triangleright$~\textit{cited (renders in Supplement):}~#1}\par\smallskip}
\newcommand{\pspine}[2]{\medskip\noindent\textbf{#1}\;\textbar\;\textit{#2}\par\nopagebreak}
% lean narrative note: ONE clause, NO "verb, role." prefix (no keep/add/demoted).
\newcommand{\nnote}[1]{\par\nopagebreak\smallskip\noindent\hspace*{2.4em}%
  \parbox{\dimexpr\linewidth-2.6em\relax}{\footnotesize\color{cNote}$\hookrightarrow$~\textbf{narrative:}~\textit{#1}}\par\smallskip}
% advisor feedback baked below the note: \pfb{status}{their comment}{our resolution}
% status in {resolved, addressing, open}; pull the comment from 3-narrative.tex \fb lines.
\newcommand{\pfb}[3]{\par\nopagebreak\noindent\hspace*{2.4em}%
  \parbox{\dimexpr\linewidth-2.6em\relax}{\footnotesize\color{cFb}$\hookrightarrow$~\textbf{Advisor}~[#1]~\emph{``#2''}~$\Rightarrow$~#3}\par\smallskip}
% sentence-points: one item = one manuscript sentence (AIM 4-5 per paragraph).
\newlist{spts}{enumerate}{1}
\setlist[spts]{label=\textsf{\scriptsize S\arabic*}, leftmargin=2.6em, itemsep=1pt, topsep=2pt}

\title{<Paper Title>}        % SHORT: the paper title only, shaped to the venue
\date{<YYYY-MM-DD>}
\begin{document}
\maketitle

\noindent\footnotesize This minimap is the paper in miniature, in the <venue>
section order. Each paragraph is concise sentence-points (S1, S2, \dots), one
point per manuscript sentence, closed by a lean gray \emph{narrative} note; where
a paragraph carries advisor feedback a maroon \textbf{Advisor} line below the note
records the comment and how we resolve it. Main-text displays render inline as
boxed thumbnails. Claim tags \cl{C1} carry the claim's content. Sources:
\texttt{2-claims.tex}, \texttt{3-narrative.tex}, \texttt{4-display.tex}; structure
per \texttt{\_venue/playbook-<venue>}.\normalsize

\medskip
\noindent\fbox{\parbox{0.97\linewidth}{\footnotesize
\textbf{Style Contract (write stage inherits this).} <venue language rules: no
em-dash; one idea per sentence; 4-5 sentences per paragraph; numbers verbatim and
source-anchored; associational vs causal posture; ... (from style-profile.md)>.}}

\hrulefill

% ===== Abstract (VENUE shape: unstructured prose OR structured labels) =======
\section*{Abstract}
\pspine{A}{phenomenon $\rightarrow$ gap $\rightarrow$ approach $\rightarrow$ findings $\rightarrow$ contribution}
\begin{spts}
\item <S1 phenomenon>.\cl{C0}
\item <S2 gap>.
\item <S3 approach / design>.\cl{C0}
\item <S4 the finding(s)>.\cl{C1}
\item <S5 the contribution>.
\end{spts}
\nnote{<one lean clause: what this paragraph must land>.}

\hrulefill

% ===== one block per section, in the VENUE'S order ==========================
% (e.g. theory-forward IMRAD for utd-is/MISQ; structured-abstract IMRAD for jama)
\section*{1.\quad <Section name>}
\pspine{<X.P1>}{<one-line job>}
\begin{spts}
\item <S1>.\cl{C1}
\item <S2>.
\item <S3>.
\item <S4>.
\end{spts}
\dcall{\textbf{<Figure/Table N>.} <caption> \cl{C1}.}
\input{0-displays/displayNN-<slug>/float.tex}
\nnote{<one lean clause>.}
\pfb{resolved}{<advisor comment, short>}{<how we resolve it>.}

% ... repeat one block per manuscript section, in venue order ...

\hrulefill

% ===== Supplement: SYNC to 0-sections + 0-Supplementary-* ===================
\section*{<n>.\quad Supplement (eAppendix, online-only)}
\noindent\footnotesize Derived from \texttt{0-sections/\{A..\}*.tex} and any
\texttt{0-Supplementary-*.tex}; keep in sync with the real SI.\normalsize
\pspine{eS.A}{<SI item>}
\begin{spts}
\item <what it contains> (\texttt{<0-sections file>}).\cl{C0}
\item Supports <main beat>.
\end{spts}
% ... one eS.* per real SI item; \wt a planned item with no draft yet ...

\hrulefill

% ===== Write-Time Reconciliations (the \wt worklist) ========================
\section*{Write-Time Reconciliations (consolidated)}
\begin{enumerate}[leftmargin=*]
\item \wt{<open item>.} <what / where it lands>.
\end{enumerate}
